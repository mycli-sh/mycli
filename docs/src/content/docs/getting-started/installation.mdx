import { Aside, LinkCard, Tabs, TabItem } from '@astrojs/starlight/components';

<Tabs>
  <TabItem label="Script (Linux/macOS)">
    ```bash
    curl -fsSL https://raw.githubusercontent.com/mycli-sh/mycli/main/scripts/install.sh | sh
    ```

    This detects your OS and architecture, downloads the latest release, and installs
    the `my` binary to `/usr/local/bin` (or `~/.local/bin` if not writable).
  </TabItem>
  <TabItem label="Homebrew">
    ```bash
    brew install mycli-sh/tap/mycli
    ```
  </TabItem>
  <TabItem label="Install with Go">
    <Aside type="note">
    mycli requires **Go 1.25+**. Make sure Go is installed and `$GOPATH/bin` is in your `PATH`.
    </Aside>

    ```bash
    go install mycli.sh/cli/cmd/my@latest
    ```

    This installs the `my` binary to your `$GOPATH/bin` directory.
  </TabItem>
  <TabItem label="Build from source">
    ```bash
    git clone https://github.com/mycli-sh/mycli.git
    cd mycli
    make build-cli
    ```

    The binary is output to `./bin/my`. You can move it anywhere in your `PATH`:

    ```bash
    sudo mv ./bin/my /usr/local/bin/my
    ```
  </TabItem>
</Tabs>

## Verify installation

```bash
my --help
```

You should see the top-level help output listing available commands.

## Shell completion

Enable tab completion for commands, arguments, and flags by generating a completion script for your shell:

```bash
# Bash
my completion bash > /etc/bash_completion.d/my

# Zsh
my completion zsh > "${fpath[1]}/_my"

# Fish
my completion fish > ~/.config/fish/completions/my.fish

# PowerShell
my completion powershell > my.ps1
```

After installing the script, restart your shell (or source the file) to activate completions. Tab completion works for command names, subcommands, and arguments like command slugs, library names, and source names.
